HibernateAbstractDAO
====================

Add in pom.xml

```
    <repository>
        <id>HibernateAbstractDAO-mvn-repo</id>
        <url>https://raw.github.com/dvdcharpentier/HibernateAbstractDAO/mvn-repo/</url>
        <snapshots>
            <enabled>true</enabled>
            <updatePolicy>always</updatePolicy>
        </snapshots>
    </repository>

```

```
  <dependency>
    <groupId>com.hibernateAbstractDAO</groupId>
		<artifactId>HibernateAbstractDAO</artifactId>
		<version>0.0.7</version>
  </dependency>
```
